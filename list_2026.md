<script>
// Load items from localStorage
function loadItems() {
    const savedItems = localStorage.getItem('list2026Items');
    if (savedItems) {
        return JSON.parse(savedItems);
    }
    // Default items
    return [
        { text: "Read 50 books (2 Completed & 2 Ongoing)", completed: false },
        { text: "Run full marathon (completed Half Marathon in 2025)", completed: True },
        { text: "Communication and StoryTelling Skills", completed: false },
        { text: "Learn to cook Desert", completed: true },
        { text: "Give trip to Mom and Dad in Hyderabad in flight", completed: false },
        { text: "Learn to drive car", completed: false },
        { text: "Learn to play flute", completed: false },
        { text: "Basic Dance moves", completed: false },
        { text: "Will talk good with people", completed: false },
        { text: "Stop Overthinking", completed: false },
        { text: "Should be more mature and not discount it with my humbleness ", completed: false },

    ];
}

// Save items to localStorage
function saveItems(items) {
    localStorage.setItem('list2026Items', JSON.stringify(items));
}

// Render the checklist
function renderChecklist() {
    const items = loadItems();
    const checklist = document.getElementById('checklist');
    
    if (items.length === 0) {
        checklist.innerHTML = `
            <div class="empty-state">
                <svg fill="currentColor" viewBox="0 0 20 20">
                    <path d="M9 2a1 1 0 000 2h2a1 1 0 100-2H9z"/>
                    <path fill-rule="evenodd" d="M4 5a2 2 0 012-2 3 3 0 003 3h2a3 3 0 003-3 2 2 0 012 2v11a2 2 0 01-2 2H6a2 2 0 01-2-2V5zm3 4a1 1 0 000 2h6a1 1 0 100-2H7zm0 4a1 1 0 000 2h6a1 1 0 100-2H7z" clip-rule="evenodd"/>
                </svg>
                <p>No items yet. Add your first goal above!</p>
            </div>
        `;
        updateProgress();
        return;
    }
    
    checklist.innerHTML = items.map((item, index) => `
        <li class="checklist-item ${item.completed ? 'completed' : ''}" data-index="${index}">
            <div class="item-number">${index + 1}</div>
            <div class="checkbox-wrapper">
                <input type="checkbox" 
                       ${item.completed ? 'checked' : ''} 
                       onchange="toggleItem(${index})"
                       id="item-${index}">
            </div>
            <label class="item-text" for="item-${index}">${escapeHtml(item.text)}</label>
        </li>
    `).join('');
    
    updateProgress();
}

// Escape HTML to prevent XSS
function escapeHtml(text) {
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}

// Toggle item completion
function toggleItem(index) {
    const items = loadItems();
    items[index].completed = !items[index].completed;
    saveItems(items);
    renderChecklist();
}

// Add new item
function addItem() {
    const input = document.getElementById('newItemInput');
    const text = input.value.trim();
    
    if (text === '') {
        alert('Please enter a task!');
        return;
    }
    
    const items = loadItems();
    items.push({ text: text, completed: false });
    saveItems(items);
    input.value = '';
    renderChecklist();
}

// Clear completed items
function clearCompleted() {
    const items = loadItems();
    const remaining = items.filter(item => !item.completed);
    
    if (remaining.length === items.length) {
        alert('No completed items to clear!');
        return;
    }
    
    if (confirm('Are you sure you want to remove all completed items?')) {
        saveItems(remaining);
        renderChecklist();
    }
}

// Update progress bar
function updateProgress() {
    const items = loadItems();
    if (items.length === 0) {
        document.getElementById('progressBar').style.width = '0%';
        document.getElementById('progressBar').textContent = '0%';
        return;
    }
    
    const completed = items.filter(item => item.completed).length;
    const percentage = Math.round((completed / items.length) * 100);
    
    document.getElementById('progressBar').style.width = percentage + '%';
    document.getElementById('progressBar').textContent = `${completed}/${items.length} (${percentage}%)`;
}

// Allow Enter key to add items
document.addEventListener('DOMContentLoaded', function() {
    document.getElementById('newItemInput').addEventListener('keypress', function(e) {
        if (e.key === 'Enter') {
            addItem();
        }
    });
    
    renderChecklist();
});
</script>
